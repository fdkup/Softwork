      $(document).ready(function () {
       $(".add").click(function (){
          $("#table1").append("<tr><td>1</td><td><input type=\"checkbox\" name=\"radio\"><td><input type=\"text\" value=\"\"/></td><td><input type=\"text\"  value=\"\"/></td><td><input type=\"text\" value=\"\"/></td></tr>");
          Sequence();
        }),
        $(".del").click(function () {
          var tr=$("#table1 tr").find("input[name='radio']:checked");
          if(tr.val()=="on"){
            tr.parent().parent().remove();
            Sequence();
          }
          else{alert("请选择要删除的项");}
        }),
         $(".up").click(function () {
           var tr=$("#table1 tr").find("input[name='radio']:checked");
           var a=tr.length;
           for(var n=0;n<a;n++) {
             var btr = tr.eq(n).parent().parent();
             var pretr = btr.prev();
             if (tr.eq(n).val() == "on") {
               var b=btr.index();
               if (b==n){
                 continue;
               }else{
                 btr.after(pretr);
                 Sequence();
               }
             } else {
               alert("请选择要上移的项")
             }
           }
         }),
         $(".down").click(function () {
           var tr=$("#table1 tr").find("input[name='radio']:checked");
           var a=tr.length;
           var c=$("#table1 tr").length;
           var m=c-1;
           for (var n=a-1;n>-1;n--){
             var btr=tr.eq(n).parent().parent();
             var pretr=btr.next();
             if(tr.eq(n).val()=="on"){
               var b=btr.index();
               if (b==m){
                 m=m-1;
               }else {
                 btr.before(pretr);
                 Sequence();
               }
             }
             else{alert("请选择要下移的项")}
           }
         })
      });
      function Sequence() {
        var len=$("#table1").find("tr").length;
        for (var i=0;i<len;i++){
          $("#table1").find("tr").eq(i).find("td").eq(0).text(i+1);
        }
      }
    </script>
